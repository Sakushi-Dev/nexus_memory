examples/basic_usage.py 
# Persistent memory in ~10 lines — offline, no model download.
from nexus_memory import NexusMemory

memory = NexusMemory()                       # one local .db file
memory.process({
    "action": "ingest",
    "interaction": {
        "query":    "where do I keep my keys?",
        "response": "In the blue ceramic bowl.",
    },
})
memory.wait()                                # ingest is async

# Recall later — by meaning, not keywords.
hits = memory.process({"action": "assemble", "query": "where are my house keys?"})
for fact in hits["raw_facts"]:
    print(f"· {fact['content']}  (importance {fact['importance']})")
memory.close()

# Layer V — a long-arc diary YOUR model writes. Offline demo.
from nexus_memory import DiaryConfig, NexusMemory

def summarize(job):                          # your secondary model goes here
    said = "; ".join(t["content"] for t in job["input"] if t["role"] == "user")
    return f"This session the user told me: {said}"

# Opt in to Layer V (off by default). One session summary every 3 ingests.
memory = NexusMemory(diary=DiaryConfig(enabled=True, update_every=3))
for query, response in [
    ("My name is Chris, I build memory systems.", "Hi Chris!"),
    ("I prefer Python; my deadline is Friday.",   "Noted."),
    ("My favorite color is purple.",              "Purple it is."),
]:
    memory.process({"action": "ingest", "interaction": {"query": query, "response": response}})
memory.wait()                                # the session summary job is now queued

applied = memory.drain_diary(summarize)      # drain the outbox on your model
print(f"drained {applied} session summary")
entry = memory.inspect(type="diary")["data"]["sessions"][-1]
print(f"diary · {entry['summary']}")
memory.close()
