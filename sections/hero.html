Nexus is an in-process, SQLite-vec backed cognitive memory module — working, episodic, semantic and procedural memory, plus an optional hierarchical diary. One process() entry point. No server. No model download.

examples/basic_usage.py 
# Long-term memory across a conversation — offline, no model download.
from nexus_memory import NexusMemory

memory = NexusMemory()                       # one local .db file

# Earlier turns in the chat — each one is remembered (Layers III + IV).
for user, assistant in [
    ("Hi! I'm Chris and I build LLM agents.",       "Nice to meet you, Chris!"),
    ("My dog Mochi is a corgi and loves the park.", "Got it — Mochi the corgi."),
    ("I keep my keys in the blue ceramic bowl.",    "Noted: the blue bowl."),
    ("Always be concise.",                          "Understood."),
]:
    memory.process({"action": "ingest", "interaction": {"query": user, "response": assistant}})
memory.wait()                                # ingests are async

# ...many turns later — recall by meaning, not keywords.
recall = memory.process({"action": "assemble", "query": "what breed is my dog?"})
for fact in recall["raw_facts"]:
    print(f"· {fact['content']}  (importance {fact['importance']})")
for rule in recall["directives"]:
    print(f"⚑ {rule}")
memory.close()

# Layer V — a first-person diary YOUR model keeps. Offline demo.
from nexus_memory import DiaryConfig, NexusMemory

def write_entry(job):                       # your model writes Nexus's prompt
    # SESSION_PROMPT tells the model: keep a first-person diary, as prose.
    return ("This session I met Chris, building a memory library in Python; "
            "his deadline is Friday and he favours purple. A friendly start.")

# Opt in to Layer V (off by default). One session entry every 3 ingests.
memory = NexusMemory(diary=DiaryConfig(enabled=True, update_every=3))
for user, reply in [
    ("Hi, I'm Chris — I build memory systems in Python.", "Nice to meet you!"),
    ("My deadline is Friday and I favour purple.",       "Noted — Friday, purple."),
    ("Wrapping up for today, thanks.",                   "Rest well, Chris."),
]:
    memory.process({"action": "ingest", "interaction": {"query": user, "response": reply}})
memory.wait()                                # the session entry job is now queued

memory.drain_diary(write_entry)               # Nexus owns the prompt; your model writes
entry = memory.inspect(type="diary")["data"]["sessions"][-1]
print(entry["summary"])                       # the assistant's first-person prose
memory.close()
